create table if not exists hospital_time(
    hospital_time_id int,
    hospital_id int,
    hospital_time_mon varchar(100) default 'null',
    hospital_time_tue varchar(100) default 'null',
    hospital_time_wed varchar(100) default 'null',
    hospital_time_thu varchar(100) default 'null',
    hospital_time_fri varchar(100) default 'null',
    hospital_time_sat varchar(100) default 'null',
    hospital_time_sun varchar(100) default 'null',
    hospital_time_mon_night boolean,
    hospital_time_tue_night boolean,
    hospital_time_wed_night boolean,
    hospital_time_thu_night boolean,
    hospital_time_fri_night boolean,
    hospital_time_sat_night boolean,
    hospital_time_sun_night boolean,
    hospital_time_holiday boolean,
    hospital_time_etc text
)ENGINE=MYISAM CHARSET=utf8;
