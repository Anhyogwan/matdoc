create table if not exists drug(
    drug_id int,
    drug_name varchar(300) default 'null',
    drug_img varchar(100) default 'null',
    drug_markf varchar(100) default 'null',
    drug_markb varchar(100) default 'null',
    drug_type varchar(20) default 'null',
    drug_colorf varchar(20) default 'null',
    drug_colorb varchar(20) default 'null',
    drug_line varchar(10) default 'null',
    drug_ingre text,
    primary key(drug_id)
)ENGINE=MYISAM CHARSET=utf8;
